Fixpoint ack (n m : nat) : nat :=
  match n with
  | O ⇒ S m
  | S p ⇒ let fix ackn (m : nat) :=
               match m with
                 | O ⇒ ack p 1
                 | S q ⇒ ack p (ackn q)
               end
           in ackn m
  end.
