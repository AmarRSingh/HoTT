Concatenation of two lists

Definition app (A : Type) : list A → list A → list A :=
  fix app l m :=
  match l with
   | nil ⇒ m
   | a :: l1 ⇒ a :: app l1 m
  end.

Infix "++" := app (right associativity, at level 60) : list_scope.
