Definition Book_9_1_15 A `{H : HoTT.categories.Category.Univalent.IsCategory A}
: IsHSet (HoTT.categories.Category.Core.object A)
  ↔ (∀ a b, IsHProp (@HoTT.categories.Category.Morphisms.Isomorphic A a b)).
Proof.
  split.
  - intros H' a b.
    eapply trunc_equiv.
    + refine (H' a b).
    + apply H.
  - intros H' a b.
    eapply trunc_equiv.
    + apply (H' a b).
    + apply (@isequiv_inverse _ _ _ (H _ _)).
Defined.
