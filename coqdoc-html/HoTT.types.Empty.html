Instance contr_from_Empty {_ : Funext} (A : Type) :
  Contr (Empty → A) :=
  BuildContr _
             (Empty_rect (fun _ ⇒ A))
             (fun f ⇒ path_forall _ f (fun x ⇒ Empty_rect (fun _ ⇒ _) x)).
